import axios from 'axios';

const API_BASE_URL = process.env.REACT_APP_API_BASE_URL || 'http://localhost:3000';

const api = axios.create({
  baseURL: API_BASE_URL,
  headers: {
    'Content-Type': 'application/json',
  },
  withCredentials: true, // Pro cookies (session)
});

// Request interceptor pro přidání tokenů
api.interceptors.request.use(
  (config) => {
    // Můžeš přidat auth token pokud budeš používat JWT místo cookies
    return config;
  },
  (error) => {
    return Promise.reject(error);
  }
);

// Response interceptor pro zpracování chyb
api.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response?.status === 401) {
      // Automatické odhlášení při 401
      localStorage.removeItem('vaultUnlocked');
      window.location.href = '/login';
    }
    return Promise.reject(error);
  }
);

export default api;
