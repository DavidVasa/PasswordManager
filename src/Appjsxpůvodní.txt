import React from 'react';
import { BrowserRouter as Router, Routes, Route, Navigate } from 'react-router-dom';
import { Toaster } from 'react-hot-toast';
import { ThemeProvider, createTheme } from '@mui/material';
import CssBaseline from '@mui/material/CssBaseline';

// Contexty
import { AuthProvider } from './context/AuthContext';

// Komponenty
import Layout from './components/Layout/Layout';

// Stránky
import LoginPage from './pages/LoginPage';
import SetupPage from './pages/SetupPage';
import DashboardPage from './pages/DashboardPage';
import ServiceTypesPage from './pages/ServiceTypesPage';
import CredentialsPage from './pages/CredentialsPage';
import NotFoundPage from './pages/NotFoundPage';

// Styly
import './styles/App.css';

// Téma
const theme = createTheme({
  palette: {
    primary: {
      main: '#1976d2',
    },
    secondary: {
      main: '#dc004e',
    },
    background: {
      default: '#f5f5f5',
    },
  },
});

// Protected Route komponenta
const ProtectedRoute = ({ children, requireUnlocked = true }) => {
  const { isAuthenticated, isVaultUnlocked, loading } = useAuth();
  
  if (loading) {
    return <div>Loading...</div>;
  }
  
  if (requireUnlocked && !isVaultUnlocked) {
    return <Navigate to="/login" />;
  }
  
  if (!requireUnlocked && isVaultUnlocked) {
    return <Navigate to="/dashboard" />;
  }
  
  return children;
};

function App() {
  return (
    <ThemeProvider theme={theme}>
      <CssBaseline />
      <AuthProvider>
        <Router>
          <div className="App">
            <Toaster position="top-right" />
            <Routes>
              {/* Public routes */}
              <Route path="/login" element={
                <ProtectedRoute requireUnlocked={false}>
                  <LoginPage />
                </ProtectedRoute>
              } />
              <Route path="/setup" element={
                <ProtectedRoute requireUnlocked={false}>
                  <SetupPage />
                </ProtectedRoute>
              } />
              
              {/* Protected routes */}
              <Route path="/" element={
                <ProtectedRoute>
                  <Layout>
                    <DashboardPage />
                  </Layout>
                </ProtectedRoute>
              } />
              <Route path="/dashboard" element={
                <ProtectedRoute>
                  <Layout>
                    <DashboardPage />
                  </Layout>
                </ProtectedRoute>
              } />
              <Route path="/service-types" element={
                <ProtectedRoute>
                  <Layout>
                    <ServiceTypesPage />
                  </Layout>
                </ProtectedRoute>
              } />
              <Route path="/credentials" element={
                <ProtectedRoute>
                  <Layout>
                    <CredentialsPage />
                  </Layout>
                </ProtectedRoute>
              } />
              
              {/* 404 */}
              <Route path="*" element={<NotFoundPage />} />
            </Routes>
          </div>
        </Router>
      </AuthProvider>
    </ThemeProvider>
  );
}

export default App;
